The easiest way to implement a multilingual toggle in Blazor is by using a dictionary-based translation service. This approach avoids heavy, complex .resx files and works universally across both Blazor Server and WebAssembly.1. Create a Localization ServiceCreate a service class to hold your translations and manage the active language.csharppublic class LocalizationService
{
    public string CurrentLanguage { get; private set; } = "en";

    private readonly Dictionary<string, Dictionary<string, string>> _translations = new()
    {
        { "en", new Dictionary<string, string> {
            { "Welcome", "Welcome to our website!" },
            { "Toggle", "Switch to Malay" }
        }},
        { "ms", new Dictionary<string, string> {
            { "Welcome", "Selamat datang ke laman web kami!" },
            { "Toggle", "Tukar ke Bahasa Inggeris" }
        }}
    };

    public string GetString(string key)
    {
        if (_translations.TryGetValue(CurrentLanguage, out var langDict) && langDict.TryGetValue(key, out var value))
            return value;
        return key; // Fallback to key if no translation exists
    }

    public void ToggleLanguage()
    {
        CurrentLanguage = CurrentLanguage == "en" ? "ms" : "en";
    }
}

Add JavaScript for Cookie HandlingBlazor cannot write HTTP cookies natively in standard client-side C# without invoking JavaScript. Add these two simple helper functions to your index.html (Blazor WebAssembly) or App.razor / _Host.cshtml (Blazor Server) inside the <body> tags:html<script>
    window.cookieInterop = {
        setCookie: function (name, value, days) {
            let expires = "";
            if (days) {
                let date = new Date();
                date.setTime(date.getTime() + (days * 24 * 60 * 60 * 1000));
                expires = "; expires=" + date.toUTCString();
            }
            document.cookie = name + "=" + (value || "") + expires + "; path=/; SameSite=Strict";
        },
        getCookie: function (name) {
            let nameEQ = name + "=";
            let ca = document.cookie.split(';');
            for (let i = 0; i < ca.length; i++) {
                let c = ca[i].trim();
                if (c.indexOf(nameEQ) == 0) return c.substring(nameEQ.length, c.length);
            }
            return null;
        }
    };
</script>
Use code with caution.2. Update the Localization ServiceModify your service to accept JavaScript runtime injection (IJSRuntime) so it can trigger the cookie save and load routines.csharpusing Microsoft.JSInterop;

public class LocalizationService
{
    private readonly IJSRuntime _js;
    public string CurrentLanguage { get; private set; } = "en";
    public event Action? OnLanguageChanged;

    private readonly Dictionary<string, Dictionary<string, string>> _translations = new()
    {
        { "en", new() { { "Welcome", "Welcome!" }, { "Toggle", "Switch Language" } } },
        { "ms", new() { { "Welcome", "Selamat Datang!" }, { "Toggle", "Tukar Bahasa" } } }
    };

    public LocalizationService(IJSRuntime js)
    {
        _js = js;
    }

    // Load cookie on initialization
    public async Task InitializeAsync()
    {
        var savedLang = await _js.InvokeAsync<string>("cookieInterop.getCookie", "BlazorCulture");
        if (!string.IsNullOrEmpty(savedLang))
        {
            CurrentLanguage = savedLang;
            OnLanguageChanged?.Invoke();
        }
    }

    public string GetString(string key) => 
        _translations.TryGetValue(CurrentLanguage, out var lang) && lang.TryGetValue(key, out var val) ? val : key;

    public async Task ToggleLanguageAsync()
    {
        CurrentLanguage = CurrentLanguage == "en" ? "ms" : "en";
        
        // Save choice to cookie for 365 days
        await _js.InvokeVoidAsync("cookieInterop.setCookie", "BlazorCulture", CurrentLanguage, 365);
        
        OnLanguageChanged?.Invoke(); 
    }
}

Update the Service for Multiple LanguagesModify your service so it accepts a specific language code (like "en", "ms", or "zh") instead of just toggling.csharpusing Microsoft.JSInterop;

public class LocalizationService
{
    private readonly IJSRuntime _js;
    public string CurrentLanguage { get; private set; } = "en";
    public event Action? OnLanguageChanged;

    // 1. Define all supported languages
    public Dictionary<string, string> SupportedLanguages => new()
    {
        { "en", "English" },
        { "ms", "Bahasa Melayu" },
        { "zh", "中文" }
    };

    private readonly Dictionary<string, Dictionary<string, string>> _translations = new()
    {
        { "en", new() { { "Welcome", "Welcome!" }, { "SelectLang", "Language" } } },
        { "ms", new() { { "Welcome", "Selamat Datang!" }, { "SelectLang", "Bahasa" } } },
        { "zh", new() { { "Welcome", "欢迎光临！" }, { "SelectLang", "语言" } } }
    };

    public LocalizationService(IJSRuntime js) => _js = js;

    public async Task InitializeAsync()
    {
        var savedLang = await _js.InvokeAsync<string>("cookieInterop.getCookie", "BlazorCulture");
        // Ensure the saved cookie matches a language we actually support
        if (!string.IsNullOrEmpty(savedLang) && SupportedLanguages.ContainsKey(savedLang))
        {
            CurrentLanguage = savedLang;
            OnLanguageChanged?.Invoke();
        }
    }

    public string GetString(string key) => 
        _translations.TryGetValue(CurrentLanguage, out var lang) && lang.TryGetValue(key, out var val) ? val : key;

    // 2. Change method to accept a specific language code
    public async Task SetLanguageAsync(string cultureCode)
    {
        if (!SupportedLanguages.ContainsKey(cultureCode)) return;

        CurrentLanguage = cultureCode;
        await _js.InvokeVoidAsync("cookieInterop.setCookie", "BlazorCulture", CurrentLanguage, 365);
        OnLanguageChanged?.Invoke(); 
    }
}
